# 9、数据库练习

学生表(student)

院系信息表(yuanxi)

  1. 查出'计算机系’的所有学生信息
# 连接查询
select t1.name, t1.sex,t1.jiguan from student t1 inner join yuanxi t2 on t1.y_id = t2.id where t2.name = '计算机系'
# 或者

# 子查询
select name, sex,jiguan from student where y_id = (select id from yuanxi where name = '计算机系')
