# 原生js实现tree的生成和筛选

jquery 实现tree展开关闭

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <title>Title</title>
  <script src="../js/jquery.min.js" type="text/javascript"></script>
  <style type="text/css">
    .menu span {
      cursor: pointer;
    }
  </style>
</head>

<body>
  <div class="menu">
  </div>
  <script>
    $(function () {
      var region = [ // 基本数据格式
        {
          children: [
            {
              children: [
                {
                  children: [],
                  code: "5001001",
                  name: "高笋塘街道",
                  pcode: "500101"
                },
                {
                  children: [],
                  code: "500101002",
                  name: "太白街道",
                  pcode: "500101"
                },
                {
                  children: [],
                  code: "500101003",
                  name: "牌楼街道",
                  pcode: "500101"
                }
              ],
              code: "500101",
              name: "万州区",
              pcdoe: "500000"
            },
            {
              children: [],
              code: "500102",
              name: "涪陵区",
              pcdoe: "500000"
            },
            {
              children: [
                {
                  children: [],
                  code: "500103001",
                  name: "七星岗街道",
                  pcode: "500103"
                },
                {
                  children: [],
                  code: "500103003",
                  name: "解放碑街道",
                  pcode: "500103"
                }
              ],
              code: "500103",
              name: "渝中区",
              pcdoe: "500000"
            }
          ],
          code: "50000",
          name: "重庆市",
          pcode: ""
        }
      ]

      let h = createTree(region)
      $(".menu").html(h)
      $(".menu span").on("click", function () {
        $(this).next().toggle();
        let code = $(this).attr('code')
        let f = filterTree()
        console.log(f(region, code))
        f = null
      });
    });

    function createTree(list) {
      let htm = '<ul>'
      for (let item of list) {
        htm += '<li>'
        htm += `<span  code="${item.code}" class="${item.children.length ? 'noLast' : 'last'}">${item.name}</span>`
        if (item.children.length > 0) {
          htm += createTree(item.children)
        }
        htm += '</li>'
      }
      htm += '</ul>'
      return htm
    }

    function filterTree() {
      let f = true
      function s(list, code) {
        let name = null
        for (let item of list) {
          if (item.code === code) {
            name = item
            f = false
            break;
          } else {
            if (f) name = s(item.children, code)
          }
        }
        return name
      }
      return s
    }
  </script>
</body>

</html>
